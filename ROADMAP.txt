  getRecruit()            — SELECT from profiles WHERE said = auth_said()
  saveRecruit()           — UPSERT profiles + trigger send-pending-account

  getShortList()          — SELECT from short_list_items WHERE said = auth_said()
  saveShortList()         — DELETE + INSERT short_list_items (owner scope)
  forgotPassword()        — supabase.auth.resetPasswordForEmail
  resetPassword()         — supabase.auth.updateUser
  requestEmailChangeMagicLink() — invoke send-email-change Edge Function
  confirmEmailChangeMagicLink() — supabase.auth.updateUser (email)
  resendSetupEmail()      — invoke send-pending-account Edge Function (resend flag)
  getSAIDStats()          — SELECT COUNT from profiles (service role key)
